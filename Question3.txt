## Pipeline example for Building using jenkins 

pipeline {
  environment {
    registry = "docker registry name"
    registryCredential = 'docker registry credentials'
    dockerImage = ''
  }
  agent any
  stages {
    stage('Cloning Git Repository') {
      steps {
        git 'https://github.com/rajat2911/SRETEST.git'
      }
    }
    stage('Building image') {
      steps{
        script {
          dockerImage = docker.build registry + ":$BUILD_NUMBER"
        }
      }
    }
    stage('Deploy Image') {
      steps{
        script {
          docker.withRegistry( '', registryCredential ) {
            dockerImage.push()
          }
        }
      }
    }
